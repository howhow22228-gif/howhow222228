        // === 1. 動態彩帶特效系統 (Confetti) ===

        const canvas = document.getElementById('confetti-canvas');

        const ctx = canvas.getContext('2d');

        let particles = [];

        let confettiAnimationId;


        function resizeCanvas() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

        }

        window.addEventListener('resize', resizeCanvas);

        resizeCanvas();


        function fireConfetti() {

            particles = [];

            const colors = ['#3b82f6', '#8b5cf6', '#ec4899', '#f59e0b', '#10b981'];

            for (let i = 0; i < 150; i++) {

                particles.push({

                    x: canvas.width / 2,

                    y: canvas.height / 2 + 100,

                    r: Math.random() * 8 + 4,

                    dx: Math.random() * 20 - 10,

                    dy: Math.random() * -15 - 5,

                    color: colors[Math.floor(Math.random() * colors.length)],

                    tilt: Math.random() * 10,

                    tiltAngle: 0,

                    tiltAngleInc: (Math.random() * 0.07) + 0.05

                });

            }

            if (!confettiAnimationId) animateConfetti();

        }


        function animateConfetti() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            let active = false;

            particles.forEach(p => {

                p.tiltAngle += p.tiltAngleInc;

                p.y += (Math.cos(p.tiltAngle) + p.dy + p.r / 2) / 2;

                p.x += Math.sin(p.tiltAngle) * 2 + p.dx;

                p.dy += 0.15; // Gravity


                if (p.y <= canvas.height) active = true;


                ctx.beginPath();

                ctx.lineWidth = p.r;

                ctx.strokeStyle = p.color;

                ctx.moveTo(p.x + p.tilt + p.r, p.y);

                ctx.lineTo(p.x + p.tilt, p.y + p.tilt + p.r);

                ctx.stroke();

            });


            if (active) {

                confettiAnimationId = requestAnimationFrame(animateConfetti);

            } else {

                cancelAnimationFrame(confettiAnimationId);

                confettiAnimationId = null;

                ctx.clearRect(0, 0, canvas.width, canvas.height);

            }

        }


        // === 2. 音效引擎 (Web Audio API) ===

        const AudioContext = window.AudioContext || window.webkitAudioContext;

        let audioCtx;


        function initAudio() {

            if (!audioCtx) audioCtx = new AudioContext();

            if (audioCtx.state === 'suspended') audioCtx.resume();

        }


        function playTone(freq, type, duration, vol, startTimeOffset) {

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            osc.type = type;

            osc.frequency.setValueAtTime(freq, audioCtx.currentTime + startTimeOffset);

            

            gain.gain.setValueAtTime(0, audioCtx.currentTime + startTimeOffset);

            gain.gain.linearRampToValueAtTime(vol, audioCtx.currentTime + startTimeOffset + 0.05);

            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + startTimeOffset + duration);

            

            osc.connect(gain);

            gain.connect(audioCtx.destination);

            osc.start(audioCtx.currentTime + startTimeOffset);

            osc.stop(audioCtx.currentTime + startTimeOffset + duration);

        }

        function initMenu() {

            const grid = document.getElementById('level-grid');

            grid.innerHTML = '';

            

            db.forEach((level) => {

                const p = progressData[level.id] || { correct: 0 };

                const total = level.items.length;

                const isCleared = p.correct === total;

                

                let statusHtml = '';

                let borderClass = 'border-slate-200';

                

                if (isCleared) {

                    statusHtml = `<span class="bg-green-100 text-green-700 font-bold px-4 py-1.5 rounded-full text-sm flex items-center gap-1.5 shadow-sm"><span class="text-lg">🏆</span> 完美通關</span>`;

                    borderClass = 'border-green-300 shadow-green-100';

                } else if (p.correct > 0) {

                    statusHtml = `<span class="text-blue-700 font-bold text-sm bg-blue-50 px-4 py-1.5 rounded-full shadow-sm border border-blue-200">挑戰中 ${p.correct}/${total}</span>`;

                    borderClass = 'border-blue-300';

                } else {

                    statusHtml = `<span class="text-slate-400 font-medium text-sm bg-slate-50 px-4 py-1.5 rounded-full">未挑戰 (${total}題)</span>`;

                }


                grid.innerHTML += `

                    <div onclick="startLevel('${level.id}')" class="bg-white/90 backdrop-blur-sm rounded-3xl p-6 shadow-lg hover:shadow-2xl cursor-pointer border-2 ${borderClass} transition-all duration-300 hover:-translate-y-2 group relative overflow-hidden">

                        <div class="absolute top-0 right-0 w-24 h-24 bg-gradient-to-br from-indigo-100 to-purple-100 rounded-full -mr-10 -mt-10 opacity-50 group-hover:scale-150 transition-transform duration-500 pointer-events-none"></div>

                    </div>

            });

        function startLevel(levelId) {

            initAudio(); // 初始化音效

            currentLevelData = db.find(l => l.id === levelId);

            activeOption = null;

            

            if (!progressData[levelId]) {

                progressData[levelId] = { correct: 0, answers: {} };

            }

            if (!progressData[levelId].answers) {

                progressData[levelId].answers = {};

            }


            document.getElementById('game-title').textContent = currentLevelData.title;

            document.getElementById('menu-view').classList.add('hidden');

            document.getElementById('game-view').classList.remove('hidden');

            window.scrollTo(0, 0);


            generateGame();

        }


        function returnToMenu() {

            document.getElementById('game-view').classList.add('hidden');

            document.getElementById('menu-view').classList.remove('hidden');

            initMenu();

        }


        function resetCurrentLevel() {

            if(confirm("確定要清除本關卡的作答紀錄重新練習嗎？")) {

                progressData[currentLevelData.id] = { correct: 0, answers: {} };

                saveProgress();

                activeOption = null;

                generateGame();

            }

        }


        // --- 動態生成 UI 避免引號衝突 ---

        function generateGame() {

            const optionsContainer = document.getElementById('options-container');

            const questionsContainer = document.getElementById('questions-container');

            

            optionsContainer.innerHTML = '';

            questionsContainer.innerHTML = '';


            const uniqueAnswers = [...new Set(currentLevelData.items.map(item => item.a))];

            uniqueAnswers.sort(() => Math.random() - 0.5); 

            

            optionsMap = uniqueAnswers.map((ans, idx) => ({

                letter: String.fromCharCode(65 + idx), 

                meaning: ans

            }));


            // 渲染上方選項按鈕

            optionsMap.forEach(opt => {

                const btn = document.createElement('button');

                btn.className = 'option-btn px-4 py-2.5 md:px-5 md:py-3 bg-white border-2 border-indigo-200 text-indigo-700 font-black text-lg md:text-xl rounded-2xl shadow-sm hover:bg-indigo-50 flex items-center gap-2.5';

                btn.id = `opt-btn-${opt.letter}`;

                

                const isUsed = isOptionFullyUsed(opt.meaning);

                if (isUsed) btn.classList.add('used');


                btn.innerHTML = `<span class="bg-indigo-100 text-indigo-800 px-2.5 py-0.5 rounded-lg shadow-sm text-base">${opt.letter}</span> <span>${opt.meaning}</span>`;

                btn.onclick = () => selectOption(opt, btn);

                optionsContainer.appendChild(btn);

            });


            // 渲染下方題目

            currentLevelData.items.forEach(item => {

                const isAnswered = progressData[currentLevelData.id].answers[item.n];

                const optForAnswer = optionsMap.find(o => o.meaning === item.a);


                const div = document.createElement('div');

                div.className = 'flex items-center text-xl py-4 border-b-2 border-slate-100 border-dashed hover:bg-blue-50/50 transition-colors rounded-xl px-3 group';

                

                // 題號

                const numSpan = document.createElement('span');

                numSpan.className = 'w-10 text-slate-400 font-black font-mono text-xl';

                numSpan.textContent = item.n + '.';

                div.appendChild(numSpan);

                

                // 括號作答區

                const blankArea = document.createElement('div');

                if (isAnswered) {

                    blankArea.className = 'blank-area filled mx-4 px-4 py-2 font-black text-green-700 flex items-center justify-center bg-green-50 border-2 border-green-300 rounded-xl shrink-0 w-[160px] md:w-[200px] shadow-sm pop-anim';

                    blankArea.innerHTML = `( <span class="tracking-widest ml-1">${optForAnswer.letter} ${item.a}</span> )`;

                } else {

                    blankArea.className = 'blank-area mx-4 px-2 py-2 text-slate-300 font-black cursor-pointer shrink-0 w-[160px] md:w-[200px] text-center hover:text-blue-500 bg-white border-2 border-slate-200 hover:border-blue-400 hover:shadow-md rounded-xl transition-all';

                    blankArea.innerHTML = `( &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; )`;

                    blankArea.onclick = () => handleBlankClick(item.n, item.a, blankArea);

                }

                div.appendChild(blankArea);


                // 題目詞語

                const titleSpan = document.createElement('span');

                titleSpan.className = 'font-black text-slate-800 tracking-wide text-2xl md:text-3xl drop-shadow-sm flex-1';

                titleSpan.textContent = item.t;

                div.appendChild(titleSpan);

                

                // 智慧注釋按鈕

                if (item.note) {

                    const noteBtn = document.createElement('button');

                    noteBtn.className = 'ml-4 px-4 py-2 bg-amber-50 text-amber-700 text-sm md:text-base font-black rounded-full border-2 border-amber-300 hover:bg-amber-100 hover:scale-105 transition-all shadow-sm shrink-0 flex items-center gap-2 group-hover:shadow-md';

                    noteBtn.innerHTML = `💡<span class="hidden sm:inline">注釋</span>`;

                    noteBtn.onclick = (e) => showAnnotation(item.t, item.note, e);

                    div.appendChild(noteBtn);

                }

                

                questionsContainer.appendChild(div);

            });

        }


        // --- 互動邏輯 ---

        function selectOption(optData, btnElement) {

            playClickSound(); // 點擊音效

            if (btnElement.classList.contains('used')) return;


            document.querySelectorAll('.option-btn').forEach(b => b.classList.remove('active'));

            

            if (activeOption && activeOption.letter === optData.letter) {

                activeOption = null;

                return;

            }


            activeOption = optData;

            btnElement.classList.add('active');

        }


        function handleBlankClick(itemNum, correctAnswer, element) {

            if (!activeOption) {

                playErrorSound();

                element.classList.add('shake-anim');

                element.innerHTML = "( 請先選上方答案 )";

                element.classList.replace('text-slate-300', 'text-rose-500');

                element.classList.replace('border-slate-200', 'border-rose-300');

                element.classList.add('bg-rose-50');

                setTimeout(() => {

                    element.classList.remove('shake-anim');

                    element.innerHTML = "( &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; )";

                    element.classList.replace('text-rose-500', 'text-slate-300');

                    element.classList.replace('border-rose-300', 'border-slate-200');

                    element.classList.remove('bg-rose-50');

                }, 800);

                return;

            }


            if (activeOption.meaning === correctAnswer) {

                // 答對邏輯

                playSuccessSound();


                progressData[currentLevelData.id].answers[itemNum] = true;

                progressData[currentLevelData.id].correct = Object.keys(progressData[currentLevelData.id].answers).length;

                saveProgress();


                element.classList.add('filled', 'pop-anim');

                element.classList.replace('text-slate-300', 'text-green-700');

                element.classList.replace('bg-white', 'bg-green-50');

                element.classList.replace('border-slate-200', 'border-green-300');

                element.classList.replace('hover:border-blue-400', 'hover:border-green-300');

                element.classList.remove('cursor-pointer', 'hover:text-blue-500', 'hover:shadow-md');

                element.onclick = null;

                element.innerHTML = `( <span class="tracking-widest ml-1">${activeOption.letter} ${correctAnswer}</span> )`;


                if (isOptionFullyUsed(correctAnswer)) {

                    document.getElementById(`opt-btn-${activeOption.letter}`).classList.add('used');

                    document.getElementById(`opt-btn-${activeOption.letter}`).classList.remove('active');

                    activeOption = null;

                }


                // 檢查是否全過關

                if (progressData[currentLevelData.id].correct === currentLevelData.items.length) {

                    setTimeout(() => {

                        playCheerSound();

                        fireConfetti(); // 觸發彩帶

                        setTimeout(() => {

                            alert("🎉 太神啦！您已完成本關卡的所有配對！\n請記得將答案填寫到講義中。");

                            returnToMenu();

                        }, 1500);

                    }, 500);

                }

            } else {

                // 答錯邏輯

                playErrorSound();

                element.classList.add('shake-anim');

                element.innerHTML = "( <span class='text-rose-600'>配對錯誤</span> )";

                element.classList.replace('border-slate-200', 'border-rose-400');

                setTimeout(() => {

                    element.classList.remove('shake-anim');

                    element.innerHTML = "( &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; )";

                    element.classList.replace('border-rose-400', 'border-slate-200');

                }, 500);

            }

        }


        function isOptionFullyUsed(meaning) {

            const itemsWithThisMeaning = currentLevelData.items.filter(item => item.a === meaning);

            const answeredItems = itemsWithThisMeaning.filter(item => progressData[currentLevelData.id].answers[item.n]);

            return itemsWithThisMeaning.length === answeredItems.length;

        }


        function saveProgress() {

            localStorage.setItem('metonymy_progress_deluxe', JSON.stringify(progressData));

        }


        // --- 全螢幕注釋邏輯 ---

        function showAnnotation(title, text, event) {

            event.stopPropagation();

            playClickSound();

            const modal = document.getElementById('annotation-modal');

            const titleEl = document.getElementById('annotation-title');

            const textEl = document.getElementById('annotation-text');

            

            titleEl.innerHTML = `<span class="bg-gradient-to-r from-indigo-600 to-purple-600 text-transparent bg-clip-text">「${title}」</span>`;

            textEl.innerHTML = text; 

            

            modal.classList.add('show');

        }


        function closeAnnotation() {

            const modal = document.getElementById('annotation-modal');

            if (modal.classList.contains('show')) {

                playClickSound();

                modal.classList.remove('show');

            }

        }
